local function S(c)
    local X = {}
    for c, V in ipairs(c:GetDescendants()) do
        if V:IsA("Attachment") and not X[V.Name] then
            X[V.Name] = V
        end
    end
    for c, V in ipairs(c:GetDescendants()) do
        if V:IsA("Beam") then
            local c = V.Attachment0 and V.Attachment0.Name
            local L = V.Attachment1 and V.Attachment1.Name
            if c and X[c] then
                V.Attachment0 = X[c]
            end
            if L and X[L] then
                V.Attachment1 = X[L]
            end
        end
    end
end

local function b(c, X, V)
    local L = c:FindFirstChild(V)
    local d = X:FindFirstChild(V)
    if not L or not d then
        return
    end
    for c, X in ipairs(L:GetChildren()) do
        if not X:IsA("WeldConstraint") then
            X:Destroy()
        end
    end
    pcall(function()
        L.Color = d.Color
        L.Material = d.Material
        L.Transparency = d.Transparency
        if L:IsA("UnionOperation") then
            L.UsePartColor = true
        end
    end)
    local i = d:Clone()
    i:PivotTo(L:GetPivot())
    for c, X in ipairs(i:GetChildren()) do
        if not X:IsA("WeldConstraint") then
            X.Parent = L
        end
    end
    i:Destroy()
    for c, X in ipairs(L:GetDescendants()) do
        if X:IsA("ParticleEmitter") then
            X.Enabled = true
            X:Clear()
        end
    end
end

local function Xi()

    ur = {}
    for c, V in pairs(c:GetPlayers()) do

            if X then
                local L, d = pcall(function() return V:GetAttribute("CurrentTable") end)
                if L and (d and d == X) then
                    c = true
                end
            end
            if not c and not X then
                local X, L = pcall(function() return V:GetAttribute("Lives") end)
                if X and typeof(L) == "number" then
                    c = true
                end
            end
            if c then
                local c, X = pcall(function() return V:GetAttribute("Lives") end)
                local L, d, i, j = nil, nil, nil, nil
                local g = V:FindFirstChild("leaderstats")
                if g then
                    local c = g:FindFirstChild("Wins")
                    local X = g:FindFirstChild("Losses")
                    local V = g:FindFirstChild("Money")
                    if c and X then
                        L = c.Value
                        d = X.Value
                        local V = L + d
                        i = V > 0 and math.floor((L / V) * 100) or 0
                    end
                    if V then
                        j = V.Value
                    end
                end
                table.insert(ur, {player = V, nama = V.Name, lives = c and X or nil, dead = false, wins = L, losses = d, winrate = i, money = j})

        end

function createTpBtn(c, X, V)
    local L = Instance.new("TextButton")
    L.Size = UDim2.new(1, 0, 0, Y and 42 or 38)
    L.LayoutOrder = V
    L.BackgroundColor3 = Color3.fromRGB(18, 16, 26)
    L.Text = c
    L.TextColor3 = Color3.fromRGB(200, 190, 220)
    L.Font = Enum.Font.GothamBold
    L.TextSize = Q
    L.Parent = Rr
    Instance.new("UICorner", L).CornerRadius = UDim.new(0, 9)
    
    local i = Instance.new("UIStroke", L)
    i.Color = Color3.fromRGB(70, 35, 130)
    i.Thickness = 1
    i.Transparency = .4
    
    local j = c
    L.MouseButton1Click:Connect(function()
        a()
        local c = d:GetAttribute("CurrentTable") ~= nil
        if c then
            L.Text = "⛔ Tidak bisa saat match!"
            L.TextColor3 = Color3.fromRGB(255, 80, 80)
            task.delay(2, function()
                L.Text = j
                L.TextColor3 = Color3.fromRGB(200, 190, 220)
            end)
            return
        end
        local V = workspace:FindFirstChild(X)
        if not V then
            L.Text = "⚠️ Folder tidak ada!"
            L.TextColor3 = Color3.fromRGB(255, 180, 50)
            task.delay(2, function()
                L.Text = j
                L.TextColor3 = Color3.fromRGB(200, 190, 220)
            end)
            print("⚠️ " .. (X .. " tidak ditemukan!"))
            return
        end
        local i
        for c, X in ipairs(V:GetDescendants()) do
            if X:IsA("BasePart") then
                i = X
                break
            end
        end
        if not i then
            print("⚠️ Tidak ada BasePart di " .. (X .. "!"))
            return
        end
        local g = d.Character and d.Character:FindFirstChild("HumanoidRootPart")
        if g then
            g.CFrame = CFrame.new(i.Position + Vector3.new(0, 5, 0))
            L.Text = "✅ TP Berhasil!"
            L.TextColor3 = Color3.fromRGB(80, 255, 120)
            task.delay(2, function()
                L.Text = j
                L.TextColor3 = Color3.fromRGB(200, 190, 220)
            end)
            print("⚡ TP ke " .. X)
        end
    end)
    
    task.spawn(function()
        while task.wait(1) do
            if not g then break end
            local c = d:GetAttribute("CurrentTable") ~= nil
            i.Transparency = c and 0 or .4
            i.Color = c and Color3.fromRGB(180, 40, 50) or Color3.fromRGB(70, 35, 130)
            L.TextColor3 = c and Color3.fromRGB(150, 140, 170) or Color3.fromRGB(200, 190, 220)
        end
    end)
    return L
end
